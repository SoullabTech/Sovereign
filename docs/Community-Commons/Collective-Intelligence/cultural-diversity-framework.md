### Cultural Diversity Calculation Methods
```typescript
// Cultural variety using Shannon diversity
calculateCulturalVariety(participants: CollectiveParticipant[]): number {
  const cultureCounts = new Map<string, number>();
  participants.forEach(p => {
    const culture = p.culturalProfile?.primaryCulture;
    if (culture) {
      cultureCounts.set(culture, (cultureCounts.get(culture) || 0) + 1);
    }
  });

  const total = participants.length;
  let diversity = 0;

  cultureCounts.forEach(count => {
    const proportion = count / total;
    diversity -= proportion * Math.log(proportion);
  });

  return diversity;
}

// Cultural bridging capacity analysis
calculateCulturalBridging(participants: CollectiveParticipant[]): number {
  let bridgingConnections = 0;
  let totalPossible = 0;

  for (let i = 0; i < participants.length; i++) {
    for (let j = i + 1; j < participants.length; j++) {
      totalPossible++;

      const culture1 = participants[i].culturalProfile?.primaryCulture;
      const culture2 = participants[j].culturalProfile?.primaryCulture;

      if (culture1 !== culture2) {
        // Cross-cultural connection potential
        const adaptability1 = participants[i].culturalProfile?.culturalAdaptability || 0;
        const adaptability2 = participants[j].culturalProfile?.culturalAdaptability || 0;

        const bridgingPotential = (adaptability1 + adaptability2) / 2;
        bridgingConnections += bridgingPotential;
      }
    }
  }

  return totalPossible > 0 ? bridgingConnections / totalPossible : 0;
}
```
