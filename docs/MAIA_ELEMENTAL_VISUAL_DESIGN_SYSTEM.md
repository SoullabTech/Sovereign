### Component State Management

```typescript
interface ElementalComponentState {
  primaryElement: Element
  phaseQuality: 'Cardinal' | 'Fixed' | 'Mutable'
  intensity: number // 0-1
  secondaryElements?: Element[]
  transitionState?: 'emerging' | 'stable' | 'dissolving'
  interactionDepth: number // 0-1, how deep in the process
}

class DisposablePixelComponent {
  private elementalState: ElementalComponentState
  private animationController: ElementalAnimationController

  updateVisualState(newPhaseState: PhaseState): void {
    this.elementalState = this.mapPhaseToVisual(newPhaseState)
    this.animationController.transition(this.elementalState)
  }

  private mapPhaseToVisual(phase: PhaseState): ElementalComponentState {
    return {
      primaryElement: phase.element,
      phaseQuality: phase.quality,
      intensity: phase.intensity,
      transitionState: this.determineTransitionState(phase),
      interactionDepth: this.calculateInteractionDepth(phase)
    }
  }
}
```
